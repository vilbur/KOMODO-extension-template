template = {'Prefset Caption': ['checkbox', 'textbox']};      var values   = {  
                        'Container A':{  
                            'Control A':    false,  
                            'Enter Text':   'Foo Text',  
                        },  
                        'Container B':{  
                            'Control B':    true,  
                            'Enter Text':   '',  
                        }  
                    };  
    paneUI().createPrefSet('#pref_set_test', template, values );  ```  __Generated prefset element__
